import express from 'express'; import path from 'path'; import fs from 'fs'; import { fileURLToPath } from 'url'; const app = express(); const PORT = process.env.PORT || 8080; // Set __dirname for ES Modules const __filename = fileURLToPath(import.meta.url); const __dirname = path.dirname(__filename); // Serve static files from the "public" folder app.use(express.static(path.join(__dirname, 'public'))); // Explicit route for the input page app.get('/input.html', (req, res) => { const inputPath = path.join(__dirname, 'public', 'input.html'); if (fs.existsSync(inputPath)) { res.sendFile(inputPath); } else { res.status(404).send("Input page not found."); } }); // Define the root route to serve main.html explicitly app.get('/', (req, res) => { const mainFile = path.join(__dirname, 'public', 'main.html'); if (fs.existsSync(mainFile)) { res.sendFile(mainFile, (err) => { if (err) { console.error("Error sending main.html:", err); res.status(500).send("Error loading page."); } }); } else { res.status(404).send("Main page not found."); } }); // Start the server app.listen(PORT, () => { console.log(`Express server is running on port ${PORT}`); });
