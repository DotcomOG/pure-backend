import express from 'express'; import path from 'path'; import fs from 'fs'; import { fileURLToPath } from 'url'; const app = express(); const PORT = process.env.PORT || 8080; // Set __dirname for ES Modules const __filename = fileURLToPath(import.meta.url); const __dirname = path.dirname(__filename); console.log("Server __dirname:", __dirname); // Serve static files from the "public" folder for main, output, etc. app.use(express.static(path.join(__dirname, 'public'))); // Route to serve the input page from the root directory app.get('/input.html', (req, res) => { // If your input.html is located in the root directory (not in "public") const inputPath = path.join(__dirname, 'input.html'); if (fs.existsSync(inputPath)) { res.sendFile(inputPath); } else { res.status(404).send("Input page not found."); } }); // Define the root route to serve main.html from the "public" folder app.get('/', (req, res) => { const mainFile = path.join(__dirname, 'public', 'main.html'); if (fs.existsSync(mainFile)) { res.sendFile(mainFile, (err) => { if (err) { console.error("Error sending main.html:", err); res.status(500).send("Error loading page."); } }); } else { res.status(404).send("Main page not found."); } }); app.listen(PORT, () => { console.log(`Express server is running on port ${PORT}`); });
