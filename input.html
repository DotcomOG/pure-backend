import express from 'express'; import path from 'path'; import fs from 'fs'; import { fileURLToPath } from 'url'; const app = express(); const PORT = process.env.PORT || 8080; // Set __dirname for ES Modules const __filename = fileURLToPath(import.meta.url); const __dirname = path.dirname(__filename); // Log __dirname for debugging console.log("Server __dirname:", __dirname); // Construct path to main.html in the public folder const mainFile = path.join(__dirname, 'public', 'main.html'); if (fs.existsSync(mainFile)) { console.log("Found main.html at:", mainFile); } else { console.log("main.html NOT found at:", mainFile); } // Serve static files from the public folder app.use(express.static(path.join(__dirname, 'public'))); // Also serve static files from the project root app.use(express.static(__dirname)); // Root route: serve main.html from the public folder app.get('/', (req, res) => { res.sendFile(mainFile, (err) => { if (err) { console.error("Error sending main.html:", err); res.status(500).send("Error loading page."); } }); }); // Explicit route for input.html (if placed in the project root) app.get('/input.html', (req, res) => { const inputFile = path.join(__dirname, 'input.html'); if (fs.existsSync(inputFile)) { res.sendFile(inputFile); } else { res.status(404).send("input.html not found"); } }); // Start the server app.listen(PORT, () => { console.log(`Express server is running on port ${PORT}`); });
